77   def loadIntoBQTable(
78                        bq: BigQuery
79                        , source_path: String
80                        , source_format: IOType
81                        , destination_dataset: String
82                        , destination_table: String
83                        , write_disposition: JobInfo.WriteDisposition
84                        , create_disposition: JobInfo.CreateDisposition
85                        , schema: Option[Schema] = None
86                      ): Map[String, Long] = {
87     // Create Output BQ table instance
88     val tableId = TableId.of(destination_dataset, destination_table)
89 
90     val jobConfiguration: JobConfiguration = source_format match {
91       case BQ => QueryJobConfiguration.newBuilder(source_path)
92         .setUseLegacySql(false)
93         .setDestinationTable(tableId)
94         .setWriteDisposition(write_disposition)
95         .setCreateDisposition(create_disposition)
96         .setAllowLargeResults(true)
97         .build()
98       case ORC | PARQUET | CSV(_,_,_,_) => schema match {
99         case Some(s) => LoadJobConfiguration
100           .builder(tableId, source_path)
101           .setFormatOptions(getFormatOptions(source_format))
102           .setSchema(s)
103           .setWriteDisposition(write_disposition)
104           .setCreateDisposition(create_disposition)
105           .build()
106         case None => LoadJobConfiguration
107           .builder(tableId, source_path)
108           .setFormatOptions(getFormatOptions(source_format))
109           .setWriteDisposition(write_disposition)
110           .setCreateDisposition(create_disposition)
111           .build()
112       }
113       case _ => throw BQLoadException("Unsupported Input Type")
114     }
115 
116     // Create BQ job
117     val jobId: JobId = JobId.of(UUID.randomUUID().toString)
118     val job: Job = bq.create(JobInfo.newBuilder(jobConfiguration).setJobId(jobId).build())
119 
120     // Wait for the job to complete
121     val completedJob = job.waitFor()
122     val destinationTable = bq.getTable(tableId).getDefinition[StandardTableDefinition]
123 
124     if (completedJob.getStatus.getError == null) {
125       load_logger.info(s"Source path: $source_path")
126       load_logger.info(s"Destination table: $destination_dataset.$destination_table")
127       load_logger.info(s"Job State: ${completedJob.getStatus.getState}")
128       load_logger.info(s"Loaded rows: ${destinationTable.getNumRows}")
129       load_logger.info(s"Loaded rows size: ${destinationTable.getNumBytes / 1000000.0} MB")
130     }
131     else {
132       throw BQLoadException(
133         s"""Could not load data in ${source_format.toString} format in table ${destination_dataset + "." + destination_table} due to error ${completedJob.getStatus.getError.getMessage}""".stripMargin)
134     }
135     Map(destination_table -> destinationTable.getNumRows)
136   }
