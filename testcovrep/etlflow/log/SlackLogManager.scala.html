43     var slackMessageForSteps = ""
44     val elapsedTime = UF.getTimeDifferenceAsString(execution_start_time, UF.getCurrentTimestamp)
45     val step_icon = if (state_status.toLowerCase() == "pass") "\n :small_blue_diamond:" else "\n :small_orange_diamond:"

47     // Update the slackMessageForSteps variable and get the information of step name and its execution time

53     // Concatenate all the messages with finalSlackMessage
