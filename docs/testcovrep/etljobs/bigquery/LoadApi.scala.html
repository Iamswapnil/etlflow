91         .build()
92     }
93 
94     // Create BQ Load job
95     val loadJob = bq.create(JobInfo.of(configuration))
96 
97     // Wait for the job to complete
98     val completedJob = loadJob.waitFor()

100

102       load_logger.info(s"Source path: $source_path")

107     }
