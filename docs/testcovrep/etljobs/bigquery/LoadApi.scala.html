92         .build()
93     }
94 
95     // Create BQ Load job
96     val loadJob = bq.create(JobInfo.of(configuration))
97 
98     // Wait for the job to complete
99     val completedJob = loadJob.waitFor()

101

108     }
109     else {
110       throw BQLoadException(
